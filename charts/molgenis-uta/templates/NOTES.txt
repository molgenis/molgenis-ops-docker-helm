To connect to the UTA on your laptop:

export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "molgenis-uta.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
kubectl port-forward $POD_NAME 15032:{{ .Values.service.port }}
export UTA_DB_URL=postgresql://anonymous@localhost:15032/uta/{{ .Values.image.tag }}
